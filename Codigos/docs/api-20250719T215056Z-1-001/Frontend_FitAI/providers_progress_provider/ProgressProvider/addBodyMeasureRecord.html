Future<bool> addBodyMeasureRecord(String type, double value) async {
  final response = await ApiService.post('progress/measure',
      json.encode({'tipo_medida': type, 'valor_cm': value}),
      token: authToken);
  if (response.statusCode == 201) {
    await fetchAllData();
    return true;
  }
  return false;
}
